155  void add_segments(size_t priority, const Polygon_2& polygon, double offset) {
156  size_t i;
157  if(segments.find(priority) == segments.end()) {
158  i=0;
159  } else {
160  i=segments.size();
161  }
162  auto orientation = polygon.orientation();
163  for(auto edge = polygon.edges_begin(); edge != polygon.edges_end(); ++edge) {
164  auto source = edge->source();
165  auto target = edge->target();
166  auto perp = (target-source).perpendicular(orientation);
167  auto len = CGAL::sqrt(CGAL::to_double(perp.squared_length()));
168  // std::cout << "len: " << len << "\n";
169  perp = offset * (perp/len);
170  target -= perp;
171  source -= perp;
172  auto v = target-source;
173  auto p_ = source + v/2;
174  auto p = Point_2(CGAL::to_double(p_.x()),CGAL::to_double(p_.y()));
175  auto l = CGAL::to_double(v.squared_length());
176  if(l<0.001) continue;
177  auto angle = std::atan2(CGAL::to_double(v.x()), CGAL::to_double(v.y()));
178  if (angle < 0) angle += pi;
179  lines.push_back(linetype(*edge, angle,p,0,priority,i++,l));
180  segments[priority].push_back(Segment_2(target,source));
181  }
182  }

218  const typename Kernel::Segment_3& a,
219  const typename Kernel::Segment_3& b,
220  std::vector<typename Kernel::Point_3>& ring_pts,
221  const float& snap_threshold,
222  const float& line_extend) {
223 
224  typedef typename Kernel::Segment_2 Segment_2;
225  typedef typename Kernel::Point_2 Point_2;
226  typedef typename Kernel::Point_3 Point_3;

228 
229  auto a_2d = Segment_2(Point_2(a.source().x(), a.source().y()), Point_2(a.target().x(), a.target().y()));
230  auto b_2d = Segment_2(Point_2(b.source().x(), b.source().y()), Point_2(b.target().x(), b.target().y()));
231 
232  auto l_a = a_2d.supporting_line();
233  auto l_b = b_2d.supporting_line();
234  Segment_2 s(a_2d.target(), b_2d.source());
235  auto result = CGAL::intersection(l_a, l_b);
236  if (result) {
237  if (auto p = boost::get<Point_2>(&*result)) {
238  if (CGAL::squared_distance(*p, s) < snap_threshold) {
239  double z = -plane.a()/plane.c() * p->x() - plane.b()/plane.c()*p->y() - plane.d()/plane.c();
240  ring_pts.push_back( Point_3(p->x(), p->y(), z) );
241  // ring_pts.push_back( plane.to_3d(*p) );
242  } else {
243  // undo any previously applied line extension prior to connected the endpoints
244  auto va = (a.target()-a.source());
245  va = va/CGAL::sqrt(va.squared_length());
246  auto vb = (b.source()-b.target());
247  vb = vb/CGAL::sqrt(vb.squared_length());
248 
249  ring_pts.push_back(a.target() - va*line_extend);
250  ring_pts.push_back(b.source() - vb*line_extend);
251  }
252  }
253  // } else if (auto l = boost::get<K::Line_2>(&*result)) {
254  } else { // there is no intersection
255  // undo any previously applied line extension prior to connected the endpoints
256  auto va = (a.target()-a.source());
257  va = va/CGAL::sqrt(va.squared_length());
258  auto vb = (b.source()-b.target());
259  vb = vb/CGAL::sqrt(vb.squared_length());
260 
261  ring_pts.push_back(a.target() - va*line_extend);
262  ring_pts.push_back(b.source() - vb*line_extend);
263  }

274  chain_ring(
275  const std::vector<size_t>& idx,
276  const typename Kernel::Plane_3& plane,
277  const std::vector<typename Kernel::Segment_3>& segments,
278  const float& snap_threshold,
279  const float& line_extend) {
280 
281  std::vector<typename Kernel::Point_3> ring_points, new_ring_points;
282 
283  if (idx.size()>1) { // we need at least 2 segments
284  for (size_t i=1; i<idx.size(); ++i) {
285  chain<Kernel>(plane, segments[idx[i-1]], segments[idx[i]], ring_points, snap_threshold, line_extend);
286  }
287  chain<Kernel>(plane, segments[idx[idx.size()-1]], segments[idx[0]], ring_points, snap_threshold, line_extend);
288 
289  // get rid of segments with zero length (ie duplicate points)
290  // check again the size, to ignore degenerate case of input ring that consists of 3 co-linear segments (would get chained to eg 0 vertices)
291  if (ring_points.size()>2) {
292  for (size_t i=1; i<ring_points.size(); ++i) {
293  check_dist<Kernel>(ring_points, new_ring_points, i-1, i);
294  }
295  check_dist<Kernel>(ring_points, new_ring_points, ring_points.size()-1, 0);
296  }
297 
298  // NOTE: at this point there can still be vertices between co-linear segments (ie 3 consecutive points on the same line)
299  }
300 
301  return new_ring_points;
302  }
