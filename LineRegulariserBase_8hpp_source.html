169  void add_segments(size_t priority, const Polygon_2& polygon,
170  double offset) {
171  size_t i;
172  if (segments.find(priority) == segments.end()) {
173  i = 0;
174  } else {
175  i = segments.size();
176  }
177  auto orientation = polygon.orientation();
178  for (auto edge = polygon.edges_begin(); edge != polygon.edges_end();
179  ++edge) {
180  auto source = edge->source();
181  auto target = edge->target();
182  auto perp = (target - source).perpendicular(orientation);
183  auto len = CGAL::sqrt(CGAL::to_double(perp.squared_length()));
184  // std::cout << "len: " << len << "\n";
185  perp = offset * (perp / len);
186  target -= perp;
187  source -= perp;
188  auto v = target - source;
189  auto p_ = source + v / 2;
190  auto p = Point_2(CGAL::to_double(p_.x()), CGAL::to_double(p_.y()));
191  auto l = CGAL::to_double(v.squared_length());
192  if (l < 0.001) continue;
193  auto angle = std::atan2(CGAL::to_double(v.x()), CGAL::to_double(v.y()));
194  if (angle < 0) angle += pi;
195  lines.push_back(linetype(*edge, angle, p, 0, priority, i++, l));
196  segments[priority].push_back(Segment_2(target, source));
197  }
198  }

233  void chain(const typename Kernel::Plane_3& plane,
234  const typename Kernel::Segment_3& a,
235  const typename Kernel::Segment_3& b,
236  std::vector<typename Kernel::Point_3>& ring_pts,
237  const float& snap_threshold, const float& line_extend) {
238  typedef typename Kernel::Segment_2 Segment_2;
239  typedef typename Kernel::Point_2 Point_2;
240  typedef typename Kernel::Point_3 Point_3;
241  typedef typename Kernel::Plane_3 Plane_3;
242 
243  auto a_2d = Segment_2(Point_2(a.source().x(), a.source().y()),
244  Point_2(a.target().x(), a.target().y()));
245  auto b_2d = Segment_2(Point_2(b.source().x(), b.source().y()),
246  Point_2(b.target().x(), b.target().y()));
247 
248  auto l_a = a_2d.supporting_line();
249  auto l_b = b_2d.supporting_line();
250  Segment_2 s(a_2d.target(), b_2d.source());
251  auto result = CGAL::intersection(l_a, l_b);
252  if (result) {
253  if (auto p = boost::get<Point_2>(&*result)) {
254  if (CGAL::squared_distance(*p, s) < snap_threshold) {
255  double z = -plane.a() / plane.c() * p->x() -
256  plane.b() / plane.c() * p->y() - plane.d() / plane.c();
257  ring_pts.push_back(Point_3(p->x(), p->y(), z));
258  // ring_pts.push_back( plane.to_3d(*p) );
259  } else {
260  // undo any previously applied line extension prior to connected the
261  // endpoints
262  auto va = (a.target() - a.source());
263  va = va / CGAL::sqrt(va.squared_length());
264  auto vb = (b.source() - b.target());
265  vb = vb / CGAL::sqrt(vb.squared_length());
266 
267  ring_pts.push_back(a.target() - va * line_extend);
268  ring_pts.push_back(b.source() - vb * line_extend);
269  }
270  }
271  // } else if (auto l = boost::get<K::Line_2>(&*result)) {
272  } else { // there is no intersection
273  // undo any previously applied line extension prior to connected the
274  // endpoints
275  auto va = (a.target() - a.source());
276  va = va / CGAL::sqrt(va.squared_length());
277  auto vb = (b.source() - b.target());
278  vb = vb / CGAL::sqrt(vb.squared_length());
279 
280  ring_pts.push_back(a.target() - va * line_extend);
281  ring_pts.push_back(b.source() - vb * line_extend);
282  }

297  std::vector<typename Kernel::Point_3> chain_ring(
298  const std::vector<size_t>& idx, const typename Kernel::Plane_3& plane,
299  const std::vector<typename Kernel::Segment_3>& segments,
300  const float& snap_threshold, const float& line_extend) {
301  std::vector<typename Kernel::Point_3> ring_points, new_ring_points;
302 
303  if (idx.size() > 1) { // we need at least 2 segments
304  for (size_t i = 1; i < idx.size(); ++i) {
305  chain<Kernel>(plane, segments[idx[i - 1]], segments[idx[i]],
306  ring_points, snap_threshold, line_extend);
307  }
308  chain<Kernel>(plane, segments[idx[idx.size() - 1]], segments[idx[0]],
309  ring_points, snap_threshold, line_extend);
310 
311  // get rid of segments with zero length (ie duplicate points)
312  // check again the size, to ignore degenerate case of input ring that
313  // consists of 3 co-linear segments (would get chained to eg 0 vertices)
314  if (ring_points.size() > 2) {
315  for (size_t i = 1; i < ring_points.size(); ++i) {
316  check_dist<Kernel>(ring_points, new_ring_points, i - 1, i);
317  }
318  check_dist<Kernel>(ring_points, new_ring_points, ring_points.size() - 1,
319  0);
320  }
321 
322  // NOTE: at this point there can still be vertices between co-linear
323  // segments (ie 3 consecutive points on the same line)
324  }
325 
326  return new_ring_points;
