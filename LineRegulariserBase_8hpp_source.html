174  void add_segments(size_t priority, const Polygon_2& polygon,
175  double offset) {
176  size_t i;
177  if (segments.find(priority) == segments.end()) {
178  i = 0;
179  } else {
180  i = segments.size();
181  }
182  auto orientation = polygon.orientation();
183  for (auto edge = polygon.edges_begin(); edge != polygon.edges_end();
184  ++edge) {
185  auto source = edge->source();
186  auto target = edge->target();
187  auto perp = (target - source).perpendicular(orientation);
188  auto len = CGAL::sqrt(CGAL::to_double(perp.squared_length()));
189  // std::cout << "len: " << len << "\n";
190  perp = offset * (perp / len);
191  target -= perp;
192  source -= perp;
193  auto v = target - source;
194  auto p_ = source + v / 2;
195  auto p = Point_2(CGAL::to_double(p_.x()), CGAL::to_double(p_.y()));
196  auto l = CGAL::to_double(v.squared_length());
197  if (l < 0.001) continue;
198  auto angle = std::atan2(CGAL::to_double(v.x()), CGAL::to_double(v.y()));
199  if (angle < 0) angle += pi;
200  lines.push_back(linetype(*edge, angle, p, 0, priority, i++, l));
201  segments[priority].push_back(Segment_2(target, source));
202  }
203  }

205  void add_segments(size_t priority, const roofer::SegmentCollection& segs) {
206  if (segs.size() == 0) return;

238  void chain(const typename Kernel::Plane_3& plane,
239  const typename Kernel::Segment_3& a,
240  const typename Kernel::Segment_3& b,
241  std::vector<typename Kernel::Point_3>& ring_pts,
242  const float& snap_threshold, const float& line_extend) {
243  typedef typename Kernel::Segment_2 Segment_2;
244  typedef typename Kernel::Point_2 Point_2;
245  typedef typename Kernel::Point_3 Point_3;
246  typedef typename Kernel::Plane_3 Plane_3;
247 
248  auto a_2d = Segment_2(Point_2(a.source().x(), a.source().y()),
249  Point_2(a.target().x(), a.target().y()));
250  auto b_2d = Segment_2(Point_2(b.source().x(), b.source().y()),
251  Point_2(b.target().x(), b.target().y()));
252 
253  auto l_a = a_2d.supporting_line();
254  auto l_b = b_2d.supporting_line();
255  Segment_2 s(a_2d.target(), b_2d.source());
256  auto result = CGAL::intersection(l_a, l_b);
257  if (result) {
258  if (auto p = boost::get<Point_2>(&*result)) {
259  if (CGAL::squared_distance(*p, s) < snap_threshold) {
260  double z = -plane.a() / plane.c() * p->x() -
261  plane.b() / plane.c() * p->y() - plane.d() / plane.c();
262  ring_pts.push_back(Point_3(p->x(), p->y(), z));
263  // ring_pts.push_back( plane.to_3d(*p) );
264  } else {
265  // undo any previously applied line extension prior to connected the
266  // endpoints
267  auto va = (a.target() - a.source());
268  va = va / CGAL::sqrt(va.squared_length());
269  auto vb = (b.source() - b.target());
270  vb = vb / CGAL::sqrt(vb.squared_length());
271 
272  ring_pts.push_back(a.target() - va * line_extend);
273  ring_pts.push_back(b.source() - vb * line_extend);
274  }
275  }
276  // } else if (auto l = boost::get<K::Line_2>(&*result)) {
277  } else { // there is no intersection
278  // undo any previously applied line extension prior to connected the
279  // endpoints
280  auto va = (a.target() - a.source());
281  va = va / CGAL::sqrt(va.squared_length());
282  auto vb = (b.source() - b.target());
283  vb = vb / CGAL::sqrt(vb.squared_length());
284 
285  ring_pts.push_back(a.target() - va * line_extend);
286  ring_pts.push_back(b.source() - vb * line_extend);
287  }
288  }

301  template <class Kernel>
302  std::vector<typename Kernel::Point_3> chain_ring(
303  const std::vector<size_t>& idx, const typename Kernel::Plane_3& plane,
304  const std::vector<typename Kernel::Segment_3>& segments,
305  const float& snap_threshold, const float& line_extend) {
306  std::vector<typename Kernel::Point_3> ring_points, new_ring_points;
307 
308  if (idx.size() > 1) { // we need at least 2 segments
309  for (size_t i = 1; i < idx.size(); ++i) {
310  chain<Kernel>(plane, segments[idx[i - 1]], segments[idx[i]],
311  ring_points, snap_threshold, line_extend);
312  }
313  chain<Kernel>(plane, segments[idx[idx.size() - 1]], segments[idx[0]],
314  ring_points, snap_threshold, line_extend);
315 
316  // get rid of segments with zero length (ie duplicate points)
317  // check again the size, to ignore degenerate case of input ring that
318  // consists of 3 co-linear segments (would get chained to eg 0 vertices)
319  if (ring_points.size() > 2) {
320  for (size_t i = 1; i < ring_points.size(); ++i) {
321  check_dist<Kernel>(ring_points, new_ring_points, i - 1, i);
322  }
323  check_dist<Kernel>(ring_points, new_ring_points, ring_points.size() - 1,
324  0);
325  }
326 
327  // NOTE: at this point there can still be vertices between co-linear
328  // segments (ie 3 consecutive points on the same line)
329  }
330 
331  return new_ring_points;
332  }
