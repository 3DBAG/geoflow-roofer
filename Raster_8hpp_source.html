102  // rasterise a polygon and return a list with points - one in the center of each pixel inside the polygon
103  // in the polygon first point is *not* repeated as last
104  // T should be a vector of arr<float,2> or arr<float,3>
105  template<typename T> std::vector<point3d> rasterise_polygon(T& polygon, std::array<double,2> cr_min, std::array<double,2> cr_max, bool returnNoData=true) const {
106  // code adapted from http://alienryderflex.com/polygon_fill/
107  int n_nodes, pixelX, pixelY, i, j, swap ;
108  int n_vertices = polygon.size();
109  std::vector<point3d> result;
110 
111 
112  // perhaps we can specialise these to the bounding box of the polygon
113  int IMAGE_TOP = std::floor(cr_min[1]),
114  IMAGE_BOT = std::ceil(cr_max[1]),
115  IMAGE_LEFT = std::ceil(cr_min[0]),
116  IMAGE_RIGHT = std::floor(cr_max[0]);
117 
118  // Loop through the rows of the image.
119  for (pixelY=IMAGE_TOP; pixelY<IMAGE_BOT; pixelY++) {
120  std::vector<int> intersect_x; // vector to hold the x-coordinates where the scanline intersects the polygon
121 
122  // Build a list of nodes.
123  n_nodes=0; j=n_vertices-1;
124  for (i=0; i<n_vertices; i++) {
125  auto pi = getColRowCoord((double)polygon[i][0], (double)polygon[i][1]);
126  auto pj = getColRowCoord((double)polygon[j][0], (double)polygon[j][1]);
127  // std::cerr << pi[0] << " " << pi[1] << "\n";
128  // std::cerr << pj[0] << " " << pj[1] << "\n";
129  if ( (pi[1]<(double) pixelY && pj[1]>=(double) pixelY)
130  || (pj[1]<(double) pixelY && pi[1]>=(double) pixelY)) {
131  intersect_x.push_back((int) (pi[0]+(pixelY-pi[1])/(pj[1]-pi[1])
132  *(pj[0]-pi[0])));
133  ++n_nodes;
134  }
135  j=i;
136  }
137 
138  // Sort the nodes, via a simple “Bubble” sort.
139  i=0;
140  while (i<n_nodes-1) {
141  if (intersect_x[i]>intersect_x[i+1]) {
142  swap=intersect_x[i]; intersect_x[i]=intersect_x[i+1]; intersect_x[i+1]=swap; if (i) i--;
143  } else {
144  i++;
145  }
146  }
147 
148  // Fill the pixels between node pairs.
149  for (i=0; i<n_nodes; i+=2) {
150  if (intersect_x[i ]>=IMAGE_RIGHT)
151  break;
152  if (intersect_x[i+1]> IMAGE_LEFT ) {
153  if (intersect_x[i ]< IMAGE_LEFT )
154  intersect_x[i ]=IMAGE_LEFT ;
155  if (intersect_x[i+1]> IMAGE_RIGHT)
156  intersect_x[i+1]=IMAGE_RIGHT;
157  for (pixelX=intersect_x[i]; pixelX<=intersect_x[i+1]; pixelX++) {
158  auto p = getPointFromRasterCoords(pixelX,pixelY);
159  if(p[2] == noDataVal_) {
160  if (returnNoData) {
161  result.push_back(p);
162  }
163  } else {
164  result.push_back(p);
165  }
166  }
167  }
168  }
169  }
170 
171  return result;
172  };
