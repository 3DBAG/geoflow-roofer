107  // rasterise a polygon and return a list with points - one in the center
108  // of each pixel inside the polygon in the polygon first point is *not*
109  // repeated as last T should be a vector of arr<float,2> or arr<float,3>
110  template <typename T>
111  std::vector<point3d> rasterise_polygon(T &polygon,
112  std::array<double, 2> cr_min,
113  std::array<double, 2> cr_max,
114  bool returnNoData = true) const {
115  // code adapted from http://alienryderflex.com/polygon_fill/
116  int n_nodes, pixelX, pixelY, i, j, swap;
117  int n_vertices = polygon.size();
118  std::vector<point3d> result;
119 
120  // perhaps we can specialise these to the bounding box of the polygon
121  int IMAGE_TOP = std::floor(cr_min[1]), IMAGE_BOT = std::ceil(cr_max[1]),
122  IMAGE_LEFT = std::ceil(cr_min[0]),
123  IMAGE_RIGHT = std::floor(cr_max[0]);
124 
125  // Loop through the rows of the image.
126  for (pixelY = IMAGE_TOP; pixelY < IMAGE_BOT; pixelY++) {
127  std::vector<int>
128  intersect_x; // vector to hold the x-coordinates where the
129  // scanline intersects the polygon
130 
131  // Build a list of nodes.
132  n_nodes = 0;
133  j = n_vertices - 1;
134  for (i = 0; i < n_vertices; i++) {
135  auto pi =
136  getColRowCoord((double)polygon[i][0], (double)polygon[i][1]);
137  auto pj =
138  getColRowCoord((double)polygon[j][0], (double)polygon[j][1]);
139  // std::cerr << pi[0] << " " << pi[1] << "\n";
140  // std::cerr << pj[0] << " " << pj[1] << "\n";
141  if ((pi[1] < (double)pixelY && pj[1] >= (double)pixelY) ||
142  (pj[1] < (double)pixelY && pi[1] >= (double)pixelY)) {
143  intersect_x.push_back(
144  (int)(pi[0] +
145  (pixelY - pi[1]) / (pj[1] - pi[1]) * (pj[0] - pi[0])));
146  ++n_nodes;
147  }
148  j = i;
149  }
150 
151  // Sort the nodes, via a simple “Bubble” sort.
152  i = 0;
153  while (i < n_nodes - 1) {
154  if (intersect_x[i] > intersect_x[i + 1]) {
155  swap = intersect_x[i];
156  intersect_x[i] = intersect_x[i + 1];
157  intersect_x[i + 1] = swap;
158  if (i) i--;
159  } else {
160  i++;
161  }
162  }
163 
164  // Fill the pixels between node pairs.
165  for (i = 0; i < n_nodes; i += 2) {
166  if (intersect_x[i] >= IMAGE_RIGHT) break;
167  if (intersect_x[i + 1] > IMAGE_LEFT) {
168  if (intersect_x[i] < IMAGE_LEFT) intersect_x[i] = IMAGE_LEFT;
169  if (intersect_x[i + 1] > IMAGE_RIGHT)
170  intersect_x[i + 1] = IMAGE_RIGHT;
171  for (pixelX = intersect_x[i]; pixelX <= intersect_x[i + 1];
172  pixelX++) {
173  auto p = getPointFromRasterCoords(pixelX, pixelY);
174  if (p[2] == noDataVal_) {
175  if (returnNoData) {
176  result.push_back(p);
177  }
178  } else {
179  result.push_back(p);
180  }
181  }
182  }
183  }
184  }
185 
186  return result;
187  };
