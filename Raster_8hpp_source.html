65  // rasterise a polygon and return a list with points - one in the center of
66  // each pixel inside the polygon in the polygon first point is *not*
67  // repeated as last T should be a vector of arr<float,2> or arr<float,3>
68  template <typename T>
69  std::vector<point3d> rasterise_polygon(T &polygon,
70  std::array<double, 2> cr_min,
71  std::array<double, 2> cr_max,
72  bool returnNoData = true) const {
73  // code adapted from http://alienryderflex.com/polygon_fill/
74  int n_nodes, pixelX, pixelY, i, j, swap;
75  int n_vertices = polygon.size();
76  std::vector<point3d> result;
77 
78  // perhaps we can specialise these to the bounding box of the polygon
79  int IMAGE_TOP = std::floor(cr_min[1]), IMAGE_BOT = std::ceil(cr_max[1]),
80  IMAGE_LEFT = std::ceil(cr_min[0]),
81  IMAGE_RIGHT = std::floor(cr_max[0]);
82 
83  // Loop through the rows of the image.
84  for (pixelY = IMAGE_TOP; pixelY < IMAGE_BOT; pixelY++) {
85  std::vector<int> intersect_x; // vector to hold the x-coordinates where
86  // the scanline intersects the polygon
87 
88  // Build a list of nodes.
89  n_nodes = 0;
90  j = n_vertices - 1;
91  for (i = 0; i < n_vertices; i++) {
92  auto pi =
93  getColRowCoord((double)polygon[i][0], (double)polygon[i][1]);
94  auto pj =
95  getColRowCoord((double)polygon[j][0], (double)polygon[j][1]);
96  // std::cerr << pi[0] << " " << pi[1] << "\n";
97  // std::cerr << pj[0] << " " << pj[1] << "\n";
98  if ((pi[1] < (double)pixelY && pj[1] >= (double)pixelY) ||
99  (pj[1] < (double)pixelY && pi[1] >= (double)pixelY)) {
100  intersect_x.push_back(
101  (int)(pi[0] +
102  (pixelY - pi[1]) / (pj[1] - pi[1]) * (pj[0] - pi[0])));
103  ++n_nodes;
104  }
105  j = i;
106  }
107 
108  // Sort the nodes, via a simple “Bubble” sort.
109  i = 0;
110  while (i < n_nodes - 1) {
111  if (intersect_x[i] > intersect_x[i + 1]) {
112  swap = intersect_x[i];
113  intersect_x[i] = intersect_x[i + 1];
114  intersect_x[i + 1] = swap;
115  if (i) i--;
116  } else {
117  i++;
118  }
119  }
120 
121  // Fill the pixels between node pairs.
122  for (i = 0; i < n_nodes; i += 2) {
123  if (intersect_x[i] >= IMAGE_RIGHT) break;
124  if (intersect_x[i + 1] > IMAGE_LEFT) {
125  if (intersect_x[i] < IMAGE_LEFT) intersect_x[i] = IMAGE_LEFT;
126  if (intersect_x[i + 1] > IMAGE_RIGHT)
127  intersect_x[i + 1] = IMAGE_RIGHT;
128  for (pixelX = intersect_x[i]; pixelX <= intersect_x[i + 1];
129  pixelX++) {
130  auto p = getPointFromRasterCoords(pixelX, pixelY);
131  if (p[2] == noDataVal_) {
132  if (returnNoData) {
133  result.push_back(p);
134  }
135  } else {
136  result.push_back(p);
137  }
138  }
139  }
140  }
141  }
142 
143  return result;
144  };
