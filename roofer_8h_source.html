35  /*
36  * @brief Configuration parameters for single instance building reconstruction
37  *
38  * //todo doc
39  */
40  struct ReconstructionConfig {
41  // control optimisation
42  float lambda = 1. / 9.;
43  // enable clipping parts off the footprint where ground planes are detected
44  bool clip_ground = true;
45  // requested LoD
46  int lod = 22;
47  // step height used for LoD13 generalisation
48  float lod13_step_height = 3.;
49  // floor elevation
50  float floor_elevation = 0.;
51  // force flat floor
52  bool override_with_floor_elevation = false;
53 
54  bool is_valid() {
55  return (lambda >= 0 && lambda <= 1.0) && lod == 12 || lod == 13 ||
56  lod == 22 && lod13_step_height > 0;
57  }
58  };

60  /*
61  * @brief Reconstructs a single instance of a building from a point cloud with
62  * one floor elevation
63  *
64  * //todo doc
65  */
66  template <typename Footprint>
67  Mesh reconstruct_single_instance(
68  const PointCollection& points_roof, const PointCollection& points_ground,
69  Footprint& footprint, ReconstructionConfig cfg = ReconstructionConfig()) {
70  try {
71  // check if configuration is valid
72  if (!cfg.is_valid()) {
73  throw rooferException("Invalid roofer configuration.");
74  }
75 
76  // prepare footprint data type
77  // template deduction will fail if not convertible to LinearRing
78  if constexpr (std::is_same_v<Footprint,
79  CGAL::Polygon_with_holes_2<EPICK>>) {
80  // convert 2D footprint to LinearRing
81  roofer::LinearRing linearRing;
82  for (auto& p : footprint.outer_boundary()) {
83  float x = p.x();
84  float y = p.y();
85  linearRing.push_back({x, y, 0.});
86  }
87  for (auto& hole : footprint.holes()) {
88  vec3f iring;
89  for (auto& p : hole) {
90  float x = p.x();
91  float y = p.y();
92  iring.push_back({x, y, 0.});
93  }
94  linearRing.interior_rings().push_back(iring);
95  }
96  cfg.override_with_floor_elevation = true;
97  }
98 
99  std::unique_ptr<roofer::reconstruction::ElevationProvider>
100  elevation_provider = nullptr;
101  if (!cfg.override_with_floor_elevation) {
102  proj_tri_util::DT base_cdt =
103  proj_tri_util::cdt_from_linearing(footprint);
104  auto base_cdt_ptr = std::make_unique<proj_tri_util::DT>(base_cdt);
105  elevation_provider =
106  roofer::reconstruction::createElevationProvider(*base_cdt_ptr);
107  } else {
108  elevation_provider = roofer::reconstruction::createElevationProvider(
109  cfg.floor_elevation);
110  }
111 
112 #ifdef ROOFER_VERBOSE
113  std::cout << "Reconstructing single instance" << std::endl;
114  std::cout << "Running plane detectors" << std::endl;
115 #endif
116  auto PlaneDetector = roofer::reconstruction::createPlaneDetector();
117  PlaneDetector->detect(points_roof);
118  if (PlaneDetector->roof_type == "no points" ||
119  PlaneDetector->roof_type == "no planes") {
120  throw rooferException(
121  "Pointcloud insufficient; unable to detect planes");
122  }
123  auto PlaneDetector_ground = roofer::reconstruction::createPlaneDetector();
124  PlaneDetector_ground->detect(points_ground);
125 
126 #ifdef ROOFER_VERBOSE
127  std::cout << "Computing alpha shapes" << std::endl;
128 #endif
129  auto AlphaShaper = roofer::reconstruction::createAlphaShaper();
130  AlphaShaper->compute(PlaneDetector->pts_per_roofplane);
131  if (AlphaShaper->alpha_rings.size() == 0) {
132  throw rooferException(
133  "Pointcloud insufficient; unable to extract boundary lines");
134  }
135  auto AlphaShaper_ground = roofer::reconstruction::createAlphaShaper();
136  AlphaShaper_ground->compute(PlaneDetector_ground->pts_per_roofplane);
137 
138 #ifdef ROOFER_VERBOSE
139  std::cout << "Running line detector" << std::endl;
140 #endif
141  auto LineDetector = roofer::reconstruction::createLineDetector();
142  LineDetector->detect(AlphaShaper->alpha_rings, AlphaShaper->roofplane_ids,
143  PlaneDetector->pts_per_roofplane);
144 
145 #ifdef ROOFER_VERBOSE
146  std::cout << "Running plane intersector" << std::endl;
147 #endif
148  auto PlaneIntersector = roofer::reconstruction::createPlaneIntersector();
149  PlaneIntersector->compute(PlaneDetector->pts_per_roofplane,
150  PlaneDetector->plane_adjacencies);
151 
152 #ifdef ROOFER_VERBOSE
153  std::cout << "Running line regulariser" << std::endl;
154 #endif
155  auto LineRegulariser = roofer::reconstruction::createLineRegulariser();
156  LineRegulariser->compute(LineDetector->edge_segments,
157  PlaneIntersector->segments);
158 
159 #ifdef ROOFER_VERBOSE
160  std::cout << "Running segment rasteriser" << std::endl;
161 #endif
162  auto SegmentRasteriser =
163  roofer::reconstruction::createSegmentRasteriser();
164  auto SegmentRasterizerCfg =
165  roofer::reconstruction::SegmentRasteriserConfig();
166  if (points_ground.empty()) {
167  SegmentRasterizerCfg.use_ground = false;
168  cfg.clip_ground = false;
169  }
170  SegmentRasteriser->compute(AlphaShaper->alpha_triangles,
171  AlphaShaper_ground->alpha_triangles,
172  SegmentRasterizerCfg);
173 
174 #ifdef ROOFER_VERBOSE
175  std::cout << "Running arrangement builder" << std::endl;
176 #endif
177  Arrangement_2 arrangement;
178  auto ArrangementBuilder =
179  roofer::reconstruction::createArrangementBuilder();
180  ArrangementBuilder->compute(arrangement, footprint,
181  LineRegulariser->exact_regularised_edges);
182 
183 #ifdef ROOFER_VERBOSE
184  std::cout << "Running arrangement optimiser" << std::endl;
185 #endif
186  auto ArrangementOptimiser =
187  roofer::reconstruction::createArrangementOptimiser();
188  ArrangementOptimiser->compute(arrangement, SegmentRasteriser->heightfield,
189  PlaneDetector->pts_per_roofplane,
190  PlaneDetector_ground->pts_per_roofplane,
191  {.data_multiplier = cfg.lambda,
192  .smoothness_multiplier = (1 - cfg.lambda),
193  .use_ground = cfg.clip_ground});
194 
195 #ifdef ROOFER_VERBOSE
196  std::cout << "Running arrangement dissolver" << std::endl;
197 #endif
198  auto ArrangementDissolver =
199  roofer::reconstruction::createArrangementDissolver();
200  ArrangementDissolver->compute(
201  arrangement, SegmentRasteriser->heightfield,
202  {.dissolve_step_edges = cfg.lod == 13,
203  .dissolve_all_interior = cfg.lod == 12,
204  .step_height_threshold = cfg.lod13_step_height});
205 
206 #ifdef ROOFER_VERBOSE
207  std::cout << "Running arrangement snapper" << std::endl;
208 #endif
209  auto ArrangementSnapper =
210  roofer::reconstruction::createArrangementSnapper();
211  ArrangementSnapper->compute(arrangement);
212 
213 #ifdef ROOFER_VERBOSE
214  std::cout << "Running arrangement extruder" << std::endl;
215 #endif
216  auto ArrangementExtruder =
217  roofer::reconstruction::createArrangementExtruder();
218  ArrangementExtruder->compute(arrangement, *elevation_provider,
219  {.LoD2 = cfg.lod == 22});
220 
221  // assert(ArrangementExtruder->meshes.size() == 1);
222  // todo temp
223  if (ArrangementExtruder->meshes.size() != 1) {
224  throw rooferException("More than one output mesh!");
225  }
226  return ArrangementExtruder->meshes.front();
227  } catch (const std::exception& e) {
228 #ifdef ROOFER_VERBOSE
229  std::cout << "Reconstruction failed, exception thrown: " << e.what()
230  << std::endl;
231 #endif
232  throw rooferException(e.what());
233  }
234  }
