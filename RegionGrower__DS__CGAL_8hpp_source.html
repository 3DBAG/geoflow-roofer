59  : points(points) {

61 
62  std::vector<point_index> indexed_points;

64 
65  size_t i = 0;
66  for (auto p : points)
67  indexed_points.push_back(
68  std::make_pair(Point(p[0], p[1], p[2]), i++));
69  Tree tree;
70  tree.insert(indexed_points.begin(), indexed_points.end());
71  neighbours.resize(size);
72 
73  for (auto pi : indexed_points) {
74  auto p = pi.first;
75  neighbours[pi.second].reserve(N);
76  Neighbor_search search(tree, p, N + 1);
77  // auto gp = glm::make_vec3(points[pi.second].data());
78  // skip the first point since it is identical to the query point
79  for (auto neighbour = search.begin() + 1; neighbour < search.end();
