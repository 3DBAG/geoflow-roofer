125  arr.unbounded_face()->data().in_footprint=false;
126  n_faces++;
127  }
128  virtual void after_split_face (Face_handle old_face,
129  Face_handle new_face, bool is_hole)
130  {
131  // Assign index to the new face.
132  if(n_faces == 1)
133  new_face->data().in_footprint = true;
134  else if(old_face->data().in_footprint) {
135  if (!is_hole && hole_mode) { // detect case where a `hole` is added that touches surrounding outer_ccb (in one vertex)
136  // these holes are ignored
137  new_face->data().in_footprint = true;
138  new_face->data().is_footprint_hole = false;
139  old_face->data().in_footprint = true;
140  old_face->data().is_footprint_hole = false;
141  std::cout << "Ignored input footprint hole that is touching footprint exterior\n";
142  } else { // normal holes that do not touch outer_ccb of existing face
143  new_face->data().in_footprint = !hole_mode;
144  new_face->data().is_footprint_hole = hole_mode;
145  }
146  } else {
147  new_face->data().in_footprint = false;
148  new_face->data().is_footprint_hole = old_face->data().is_footprint_hole;
149  }
150  n_faces++;
151  }
152  void set_hole_mode(bool mode) {
153  hole_mode = mode;
154  }
