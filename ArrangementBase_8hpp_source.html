47  FaceInfo operator()(const FaceInfo a, const FaceInfo b) const {
48  auto r = FaceInfo();
49  r.segid = 0;
50  // if (a.is_finite && b.is_finite)
51  r.is_finite = true;
52 
53  if (a.segid != 0 && b.segid == 0) {
54  r = a;
55  } else if (a.segid == 0 && b.segid != 0) {
56  r = b;
57  } else if (a.segid != 0 &&
58  b.segid != 0) { // we need to merge 2 faces with a plane
59  if (a.elevation_70p > b.elevation_70p) {
60  r = a;
61  } else {
62  r = b;
63  }
64  }
65 
66  if (a.in_footprint || b.in_footprint) {
67  r.in_footprint = true;
68  }
69 
70  return r;
71  }

127  virtual void after_split_face(Face_handle old_face, Face_handle new_face,

129  // Assign index to the new face.
130  if (n_faces == 1)
131  new_face->data().in_footprint = true;
132  else if (old_face->data().in_footprint) {
133  if (!is_hole &&
134  hole_mode) { // detect case where a `hole` is added that touches
135  // surrounding outer_ccb (in one vertex)
136  // these holes are ignored
137  new_face->data().in_footprint = true;
138  new_face->data().is_footprint_hole = false;
139  old_face->data().in_footprint = true;
140  old_face->data().is_footprint_hole = false;
141  std::cout << "Ignored input footprint hole that is touching footprint "
142  "exterior\n";
143  } else { // normal holes that do not touch outer_ccb of existing face
144  new_face->data().in_footprint = !hole_mode;
145  new_face->data().is_footprint_hole = hole_mode;
146  }
147  } else {
148  new_face->data().in_footprint = false;
149  new_face->data().is_footprint_hole = old_face->data().is_footprint_hole;
150  }
151  n_faces++;
152  }
153  void set_hole_mode(bool mode) { hole_mode = mode; }
