42  // Note this crashes when idx.size()==1;
43  inline double fit_plane(std::vector<size_t>& idx, Plane& plane){
44  std::vector<Point> neighbor_points;
45  for (auto i: idx)
46  neighbor_points.push_back(Point(points[i][0], points[i][1], points[i][2]));
47  double quality = linear_least_squares_fitting_3(neighbor_points.begin(),neighbor_points.end(),plane,CGAL::Dimension_tag<0>());
48  return quality;
49  }

51  virtual std::deque<size_t> get_seeds() override {
52  // seed generation
53  typedef std::pair<size_t,double> index_dist_pair;
54  auto cmp = [](index_dist_pair left, index_dist_pair right) {return left.second < right.second;};
55  std::priority_queue<index_dist_pair, std::vector<index_dist_pair>, decltype(cmp)> pq(cmp);
56 
57  size_t i=0;
58  Plane plane;
59  for(size_t pi=0; pi<size; ++pi){
60  auto neighbours = get_neighbours(pi);
61  auto quality = fit_plane(neighbours, plane);
62  pq.push(index_dist_pair(i++, quality));
63  }
64 
65  std::deque<size_t> seed_order;
66  while (pq.size()>0) {
67  seed_order.push_back(pq.top().first); pq.pop();
68  }
69  return seed_order;
70  }

87  dist_thres(dist_thres), normal_thres(normal_thres), n_refit(n_refit) {};
88 
89  bool is_valid(PlaneDS& cds, size_t candidate, size_t neighbour, PlaneRegion& shape) {
90  Point p_c = Point(cds.points[candidate][0], cds.points[candidate][1], cds.points[candidate][2]);
91  Vector n_c = Vector(cds.normals[candidate][0], cds.normals[candidate][1], cds.normals[candidate][2]);
92  Point p = Point(cds.points[neighbour][0], cds.points[neighbour][1], cds.points[neighbour][2]);
93  Vector n = Vector(cds.normals[neighbour][0], cds.normals[neighbour][1], cds.normals[neighbour][2]);
94 
95  if (shape.inliers.size()==0) {
96  shape.plane = Plane(p_c,n_c);
97  shape.inliers.push_back(candidate);
98  }
99 
100  bool valid = (CGAL::squared_distance(shape.plane, p) < dist_thres) && (std::abs(shape.plane.orthogonal_vector()*n) > normal_thres);
101  if (valid) {
102  shape.inliers.push_back(neighbour);
103  if (shape.inliers.size() % n_refit ==0) {
104  cds.fit_plane(shape.inliers, shape.plane);
105  }
106  }
107  return valid;
108  }
109 };
