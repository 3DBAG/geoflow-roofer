49  virtual std::deque<size_t> get_seeds() override {
50  // seed generation
51  typedef std::pair<size_t,double> index_dist_pair;
52  auto cmp = [](index_dist_pair left, index_dist_pair right) {return left.second < right.second;};
53  std::priority_queue<index_dist_pair, std::vector<index_dist_pair>, decltype(cmp)> pq(cmp);
54 
55  size_t i=0;
56  Plane plane;
57  for(size_t pi=0; pi<size; ++pi){
58  auto neighbours = get_neighbours(pi);
59  auto quality = fit_plane(neighbours, plane);
60  pq.push(index_dist_pair(i++, quality));
61  }
62 
63  std::deque<size_t> seed_order;
64  while (pq.size()>0) {
65  seed_order.push_back(pq.top().first); pq.pop();
66  }
67  return seed_order;
68  }

87  bool is_valid(PlaneDS& cds, size_t candidate, size_t neighbour, PlaneRegion& shape) {
88  Point p_c = Point(cds.points[candidate][0], cds.points[candidate][1], cds.points[candidate][2]);
89  Vector n_c = Vector(cds.normals[candidate][0], cds.normals[candidate][1], cds.normals[candidate][2]);
90  Point p = Point(cds.points[neighbour][0], cds.points[neighbour][1], cds.points[neighbour][2]);
91  Vector n = Vector(cds.normals[neighbour][0], cds.normals[neighbour][1], cds.normals[neighbour][2]);
92 
93  if (shape.inliers.size()==0) {
94  shape.plane = Plane(p_c,n_c);
95  shape.inliers.push_back(candidate);
96  }
97 
98  bool valid = (CGAL::squared_distance(shape.plane, p) < dist_thres) && (std::abs(shape.plane.orthogonal_vector()*n) > normal_thres);
99  if (valid) {
100  shape.inliers.push_back(neighbour);
101  if (shape.inliers.size() % n_refit ==0) {
102  cds.fit_plane(shape.inliers, shape.plane);
103  }
104  }
105  return valid;
106  }
